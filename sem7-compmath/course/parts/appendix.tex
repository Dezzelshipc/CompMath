\section{Приложения}

\subsection{Примеры задач} \label{examples}

\begin{itemize}
    \item Задача 1:
    \[
    \begin{cases}
        y'' + y = -x; \\
        y(0) = 0, \quad y(1) = 0; \\
        y(x) = \dfrac{\sin x}{\sin 1} - x
    \end{cases}
    \]\label{z1}

    \item Задача 2:
    \[
    \begin{cases}
        y'' - y = -x; \\
        y(0) = 1, \quad y(1) = e + 1; \\
        y(x) = x + e^x
    \end{cases}
    \]\label{z2}
    
    \item Задача 3:
    \[
    \begin{cases}
        y'' - \dfrac{2y}{(x+1)^2} = \dfrac{4.5}{(x+1)^{3/2}}; \\
        y(0) - 2y'(0) = 0, \quad y'(1) = -\dfrac{1}{\sqrt{2}}; \\
        y(x) = -2\sqrt{x+1}
    \end{cases}
    \]\label{z3}
\end{itemize}
